# Python Objects

```
# A string object
my_string = "Hello, world!"

# An integer object
my_integer = 10

# A list object
my_list = [1, 2, 3, 4, 5]

# A dictionary object
my_dictionary = {"name": "Alice", "age": 25}

# A function object
def my_function(x):
  return x * 2

# A class object
class MyClass:
  def _init_(self, name):
    self.name = name

  def greet(self):
    print("Hello, my name is {}".format(self.name))

# An instance of a class object
my_instance = MyClass("Alice")

# Calling a method on an object
my_instance.greet()

```
